CREATE TABLE tm_ms_user (
  USER_ID varchar(50) NOT NULL,
  PASSWORD varchar(10) NOT NULL,
  USER_NAME varchar(50) NOT NULL,
  PRIMARY KEY (USER_ID),
  UNIQUE KEY USER_ID (USER_ID)
)


CREATE TABLE tm_tr_lecturer (
  LECNUMBER varchar(10) NOT NULL,
  NAME varchar(50) NOT NULL,
  FURIGANA varchar(50) NOT NULL,
  STARTLECTURER varchar(10) NOT NULL,
  PRIMARY KEY (LECNUMBER),
  UNIQUE KEY LECNUMBER (LECNUMBER)
)


CREATE TABLE tm_tr_trainee (
  PCNUMBER varchar(10) NOT NULL,
  NAME varchar(50) NOT NULL,
  FURIGANA varchar(50) NOT NULL,
  NEWHALFWAY varchar(1) NOT NULL,
  TRAININGSTART varchar(10) NOT NULL,
  STE2PASSE varchar(2) DEFAULT NULL,
  COUNT int NOT NULL,
  LECTURERNAME varchar(10) NOT NULL,
  GENDER varchar(1) NOT NULL,
  PRIMARY KEY (PCNUMBER),
  UNIQUE KEY PCNUMBER (PCNUMBER)
)
